COMMAND-NAME = MEANING

cls = לנקות את המסך מפקודות
dir = לראות את הקבצים התיקיות בסיפרייה ומידע על הזיכרון הפנוי
cd.. = חוזר תיקייה את אחורה
cd/ = לחזור לתיקייה הראשונה( C )
cd/FOLDER-NAME/FOLDER-NAME = להיכנס לתוך תיקייה ספציפית
cd FOLDER-NAME = להיכנס לתוך תיקייה
md FOLDER-NAME = ליצור תיקייה חדשה
tree = מראה את "עץ" הסיפריות(תיקיות) שיש במיקום הנוכחי
help COMMAND או COMMAND /? = כדי להציג פעולות אפשריות לפקודה
dir /p = להראות קבצים באורך המסך(לחיצה על כפתור יביא את המשך הקבצים)
dir /o:n =  להראות קבצים בסדר מסויים (/o), לפי סדר השמות(:n)
dir /o:s =  להראות קבצים בסדר מסויים (/o), לפי גודל הקבצים(:s)
 rmdir /s FOLDER-NAME = למחוק תיקייה
COMMAND-NAME > FILE-NAME.txt = לכתוב את התוצאה בקובץ טקסט(dir > abc.txt ירשום את רשימת הקבצים לתוק קובץ ששמו abc)
copy FILE-NAME DESTINATION-ADDRESS = להעתיק קובץ למיקום מסויים
FILE-NAME DESTINATION-ADDRESS/NEW-NAME = להעתיק למיקום מסויים ולשנות את שם הקובץ
del FILE-NAME = למחוק קובץ
move FILE-NAME DESTINATION-ADDRESS = להעביר קובץ למיקום מסויים
type FILE-NAME = מציג את תכולת הקובץ
attrib FILE-NAME = מציג את מאפיין הקובץ(רגיל,לקריאה בלבד,נסתר)
attrib +R FILE-NAME = להפוך את הקובץ לקריאה בלבד
attrib -R FILE-NAME = לבטל את הקובץ מקריאה בלבד
attrib +H FILE-NAME = להפוך את הקובץ לנסתר(מינוס במקום הפלוס כדי להוריד)
dir FILE-NAME = פרטים על קובץ ספציפי
dir * = להציג הכל(כוכבית נקודה טקסט יביא את כל קבצי הטקסט)
dir ? = יציג הכול במקום הסימן שאלה רק על תו אחד בודד
dir B?.* = יביא את כל הקבצים שמתחילים באות בי  ויש עוד אות אחת עם כל סיומת אפשרית
